cd mis-programas-python/"JUEGOS_DE_TEXTO_ PLANETAS _BIBLIOTECA_colorama"

# Instalar dependencias

# Ejecutar un juego

# O para ver el índice HTML (este archivo)
# Abrir index.html en el navegador
